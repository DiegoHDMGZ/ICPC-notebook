typedef long long ll;

struct Segment{
	ll left,right;
};

ll segmentUnion(const vector<pair <ll,ll> > &a){ //O(nlogn)
	ll n = a.size();
	
	vector< pair< ll, bool > > x(n*2);
	
	FOR(i,n){
		x[i*2] = make_pair(a[i].first , false);
		x[i*2+1] = make_pair(a[i].second,true);
	}
	
	sort(x.begin(),x.end());
	
	ll result = 0;
	ll c = 0;
	
	FOR(i,2*n){
		if(c > 0){
			result += (x[i].first -x[i-1].first);
		}
		
		if(x[i].second) c++;
		else c--;
		
		return result;
	}
}
