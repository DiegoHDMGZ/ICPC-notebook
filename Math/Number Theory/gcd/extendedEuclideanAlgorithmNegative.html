using namespace std;

ll mod(ll a, ll b){
    ll r = a%b;
    if(r < 0) r+= b;
    return r;
}

ll divEnt(ll a, ll b){
    return (a-mod(a,b))/b;
}

ll gcd (ll a, ll b, ll & x, ll & y) { //O(max(loga,logb))

    if (a == 0) {
        if(b < 0){
            x = 0;
            y = -1;
            return -b;
        }
        x = 0;
        y = 1;
        return b;
    }
    ll x1, y1;
    ll d = gcd (mod(b,a), a, x1, y1);
    x = y1 - divEnt(b,a) * x1;
    y = x1;
    return d;
}
