#define snum(x) scanf("%lld",&x)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0)
#define MAX 5000
#define INF (ll)(1e15)
using namespace std;

typedef long long ll;

struct Graph{
	vector<ll> adj[MAX];
	ll cap[MAX][MAX]; //capacidad
	bool vis[MAX];
	bool M[MAX][MAX];
	
	void build(){
		FOR(i,MAX){
			adj[i].clear();
			FOR(j,MAX){
				cap[i][j] = 0;
				M[i][j] = false;
				vis[i] = false;
			}
		}
	}
	
	ll dfs(ll u, ll t ,ll f){
		//t = target
		if(u == t) return f;
		if(vis[u]) return 0;
		vis[u] = true;
		
		FOR(i,adj[u].size()){
			ll v = adj[u][i];
			if(cap[u][v] == 0) continue;
			
			ll ret = dfs(v,t, min(f,cap[u][v]) );
			
			if(ret > 0){
				cap[u][v] -= ret;
				cap[v][u] += ret;
				return ret;
			}
		}
		return 0;
	}
	
	ll fordFulkerson(ll s, ll t){ //O((n+m)|F|
		ll flow = 0;
		while(true){
			ms(vis,false,MAX);
			ll inc = dfs(s,t,INF);
			if(inc == 0) break;
			flow += inc;
		}
		return flow;
	}
	
	void addEdge(ll u, ll v, ll uv, ll vu = 0){
		if(!M[u][v]) adj[u].pb(v);
		if(!M[v][u]) adj[v].pb(u);
		
		M[u][v] = M[v][u] = true;
		cap[u][v] += uv;
		cap[v][u] += vu;
	}
} G;
