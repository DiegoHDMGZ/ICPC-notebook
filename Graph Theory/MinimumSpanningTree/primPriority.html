#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define FOR(i,n) for(ll i = 0; i < n; i++)
#define pb push_back
#define ms(a,valor,n) FOR(i,n) a[i] = valor;
#define snum(x) scanf("%lld",&x)
#define INF (ll)(1e15)
#define MAX 100000
using namespace std;

typedef long long ll;

struct Edge{
    ll w ,nodo;
    Edge(){
        w = INF;
        nodo = -1;
    }
    
    Edge(ll t, ll _w){
        w = _w;
        nodo = t;
    }
    
    bool operator <(Edge const &other) const{
        return w > other.w;
    }
};

ll n;
vector<ll>adj [MAX];
vector<ll> peso[MAX];

ll minPeso[MAX];
ll p[MAX];
bool selected[MAX];

ll prim(){ //O(mlogn)
    FOR(i,MAX){
    	p[i] = -1;
    	minPeso[i] = INF;
    	selected[i] = false;
	}
    ll totalWeight = 0;
	minPeso[0] = 0;
    priority_queue<Edge> q;
    
    Edge ini(0,0);
    q.push(ini);
    
    ll totalNodos = 0;
	while(!q.empty()){
	    //encontramos la arista siguiente con menor peso
	    Edge e = q.top();
	    q.pop();
	    ll u = e.nodo;
	    if(selected[u]) continue;
	    
	    totalWeight += e.w;
	    selected[u] = true;
	    totalNodos++;
	
	    FOR(j,adj[u].size()){
	        ll v = adj[u][j];
	        if(peso[u][j] < minPeso[v] && !selected[v]){
	            minPeso[v] = peso[u][j];
	            p[v] = u;
				q.push(Edge(v,peso[u][j]));
	        }
	    }
	}
	
	if(totalNodos != n) return -1;
    return totalWeight;
}
