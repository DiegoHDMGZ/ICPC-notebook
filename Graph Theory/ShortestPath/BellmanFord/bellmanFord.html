#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define ll long long
#define FOR(i,n) for(int i = 0; i < n; i++)
#define pb push_back
#define ms(a,valor,n) FOR(i,n) a[i] = valor;
#define snum(x) scanf("%lld",&x);
#define MAX 10000
#define INF (ll)1e10
using namespace std;

ll d[MAX];
ll p[MAX];

struct Edge{
	ll u,v,cost;
};

Edge E[MAX];

ll n,m;
bool bellmanFord(ll root){ //O(nm)
	ms(d,INF,MAX);
	ms(p,-1,MAX);
	
	d[root] = 0;

	ll x;
	FOR(i,n){
		x = -1; //indicador de ciclo negativo
		bool mejora = false;
		FOR(j,m){
			if(d[E[j].u] < INF){
				if(d[E[j].v] > d[E[j].u] + E[j].cost){
					d[E[j].v] = max(-INF,d[E[j].u] + E[j].cost); //para evitar el overflow
					p[E[j].v] = E[j].u;
					x = E[j].v;
					mejora = true;
				}
			}
		}
		
		if(!mejora) break;
	}
	
	if(x == -1){
		return false; //no hay ciclo negativo
	}
	else{
		return true;
		ll y = x;
		
		FOR(i,n){
			y = p[y]; //retrocedemos por si acaso n veces. No hay perdida porque es un ciclo
		}
		
		deque<ll> path;
		for(ll actual = y; ; actual = p[actual]){
			path.push_front(actual);
			if(actual == y && path.size() > 1){
				break;
			}
		}
	}

}

int main() {
	snum(n); //# de nodos
	snum(m); //# de aristas
	
	FOR(i,m){
		//only directed graphs
		snum(E[i].u);
		E[i].u--;
		snum(E[i].v);
		E[i].v--;
		snum(E[i].cost);
	}
	return 0;
}
