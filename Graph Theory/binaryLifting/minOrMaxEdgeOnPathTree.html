#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define FOR(i,n) for(ll i = 0; i < n; i++)
#define pb push_back
#define ms(a,valor,n) FOR(i,n) a[i] = valor;
#define snum(x) scanf("%lld",&x)
#define FAST ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0)
#define INF (ll)(0)
#define MAX 100000
using namespace std;

typedef long long ll;

ll const loga = log2(MAX)+1;

struct Graph{
	vector<ll>adj [MAX];
	vector<ll> cost[MAX];
	ll nGraph;
	
	ll p[MAX];
	ll pPeso[MAX];
	ll anc[MAX][loga];
	ll L[MAX]; // iniciar L[ ] en -1
	ll st[MAX][loga];
	
	void dfs(ll u, ll nivel){ //O(n + m)
		L[u] = nivel;
	
		FOR(i,adj[u].size()){
			ll v = adj[u][i];
			if(L[v] == -1){
				dfs(v,nivel+1);
				p[v] = u;
				pPeso[v] = cost[u][i];
			}
		}
	}
	
	ll f(ll a, ll b){
		return max(a,b); //cambiar para min
	}
	
	void precalculate(ll root){ //O(nlogn)
		ms(L,-1,MAX);
		dfs(root,0);
		FOR(i,nGraph){
			for(ll j = 0; (1LL << j) < nGraph; j++){
				anc[i][j] = -1;
			}	
		}
		
		for(int i = 0; i < nGraph; i++){
			anc[i][0] = p[i];
			st[i][0] = pPeso[i];
		}
		
		for(int j = 1; (1 << j) < nGraph; j++){
			for(int i = 0; i < nGraph; i++){
				if(anc[i][j-1] != -1){
					anc[i][j] = anc[anc[i][j-1]][j-1];
					st[i][j] = f(st[i][j-1],st[anc[i][j-1]][j-1]);
				}
			}
		}
	}
	
	ll maxOnPath(ll u, ll v){ //O(logn)
		if(L[u] < L[v]){
			swap(u,v);
		}
		
		int lg = 31 - (__builtin_clz(L[u]));
		
		ll maxU,maxV;
		maxU = maxV = INF; //cambiar para min
		for(int i = lg; i >= 0; i--){
			if(L[u] - (1 << i) >= L[v]){
	
				maxU = f(maxU,st[u][i]);
	
				u = anc[u][i];
			
			}
		}
		
		if(u == v){
			return maxU;
		}
		
		for(int i = lg; i >= 0; i--){
			if(anc[u][i] != -1 && anc[u][i] != anc[v][i]){
				maxU = f(maxU,st[u][i]);
				maxV = f(maxV,st[v][i]);
				u = anc[u][i], v = anc[v][i];
			}
		}
		return f( f(maxU,maxV) , f(pPeso[u],pPeso[v]) );
	}
} G;
