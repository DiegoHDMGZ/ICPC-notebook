#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define ll long long
#define FOR(i,n) for(int i = 0; i < n; i++)
#define pb push_back
#define MAX 100000
#define ll long long
using namespace std;

ll n,m;

ll A[MAX],B[MAX];

ll f(ll fijo){
	ll cont = 0;
	
	FOR(i,n){
		cont += max(0LL,fijo-A[i]);
	}
	
	FOR(i,m){
		cont += max(0LL,B[i]-fijo);
	}
	return cont;
}

ll ternarySearch(ll ini, ll fin){ //O(log x * O(f))
	while(fin - ini > 2){
		ll m1 = ini + (fin-ini)/3;
		ll m2 = fin - (fin-ini)/3;
		if(f(m1) > f(m2)){
			ini = m1;
		}
		else{
			fin = m2;
		}
	}
	
	ll mini = f(ini);
	for(int i = ini; i <= fin; i++){
		mini = min(mini,f(i));
	}
	
	return mini;
}

int main() {
	cin >> n >> m;
	FOR(i,n) cin >> A[i];
	FOR(i,m) cin >> B[i];
	
	cout << ternarySearch(1,1e9) << endl;
	
	return 0;
}
