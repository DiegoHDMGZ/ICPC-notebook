void debugDeque( deque<ll> d){
	for(ll i = 0; i < d.size() ; i++){
		cout << d[i];
	}
	cout << endl;
}

deque<ll> suma(deque<ll> a, deque<ll> b){
	ll llevo = 0;
	deque<ll>  resultado;
	ll i = a.size( ) - 1, j = b.size() - 1;
	
	while(i >= 0 && j >= 0){
		ll suma = a[i] + b[j]  + llevo;
		llevo = 0;
		if( suma >= 10){
			llevo = 1;
			suma = suma % 10;
		}
		resultado.push_front(suma);
		i--;
		j--;
	}
	
	if( i >= 0 ){
		for(ll k = i; k >= 0 ; k--){
			ll suma = a[k] + llevo;
			llevo = 0;
			if( suma >= 10){
				llevo = 1;
				suma = suma % 10;
			}
			resultado.push_front(suma);
		}
	}
	else{
		for(ll k = j ; k >= 0; k--){
			ll suma = b[k] + llevo;
			
			llevo = 0;
			
			if( suma >= 10){
				llevo = 1;
				suma = suma % 10;
			}
			resultado.push_front(suma);
		}
	}
	
	if( llevo > 0 ){
		resultado.push_front(1);
	}
	return resultado;
}

deque<ll> multUnDigito(deque<ll> a, ll c){
	if(c == 0){
		deque<ll> resp;
		resp.push_front(0);
		return resp;
	}
	
	if(c == 1){
		return a;
	}
	
	deque<ll> resultado;
	ll llevo = 0;
	
	for(ll i = a.size() - 1; i >=0; i--){
		ll mult = a[i] * c + llevo;
		
		llevo = 0;
		
		if(mult >= 10){
			llevo = mult / 10;
			mult = mult % 10;
		}
		
		resultado.push_front(mult);
	}
	
	if(llevo > 0){
		resultado.push_front(llevo);
	}
	return resultado;
}

deque<ll> mult10(deque<ll> num, ll pot){
	for(ll i = 0; i < pot; i++){
		num.push_back(0);
	}
	return num;
}

deque<ll> mult(deque<ll> a, deque<ll> b){
	deque<ll> resultado;
	
	for(ll i = b.size() - 1; i >= 0; i--){
		ll pot = b.size() - i - 1;
		deque<ll> auxiliar = multUnDigito(a , b[i]);
		auxiliar = mult10(auxiliar, pot);
		resultado = suma(resultado, auxiliar);
	}
	
	return resultado;
}

int main() {
	deque<ll> t1, t2;
	ll a,b,n;

	t1.push_back( a);

	for(ll i = 3; i <= n; i++){
		deque<ll> auxiliar = t2;
		t2 = suma(t1, mult(t2, t2));
		t1 = auxiliar;
	}

}
