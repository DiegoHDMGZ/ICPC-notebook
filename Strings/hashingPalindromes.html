#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define ll long long
#define FOR(i,x) for(int i = 0; i < x ; i++)
#define pb push_back
#define MAX 5000000

using namespace std;

ll B = 67; //Utilizamos el caso extremo en que se trabajo con digitos, letras ( minus y mayus)
//en caso se quiera restringir, cambiar en todos los hashing 
ll mod = 1e9+7;
ll pot[MAX];
ll hPref[MAX];
ll hSuf[MAX];
int minimoCaracter = (int)('0');

string s;

ll hSub(int i, int j){ //O(1)
	if(i == 0) return hPref[j];
	return ( hPref[j]-(hPref[i-1]*pot[j-i+1] ) % mod+mod)%mod;
}

ll hSub2(int i, int j){ //O(1)
	if(j == s.size()-1) return hSuf[i];
	return (hSuf[i]-(hSuf[j+1]*pot[j-i+1] ) % mod+mod)%mod;
}

bool esPalindroma(int i, int j){ //O(1)
	
	if(i == j) return true;
	int limInf = (j-i-1)/2;
	limInf += i;
	int limSup = (j-i+1)%2 == 0 ? limInf+1 : limInf+2;
	
	if(hSub(i,limInf) == hSub2(limSup,j)){
		return true;
	}
	
	return false;
}

void precalc(string &s){ //O(size)
	hPref[0] = s[0]-minimoCaracter+1;

    for(int i = 1; i < s.size(); i++){
    	hPref[i] = ( (hPref[i-1]*B) % mod+(s[i]-minimoCaracter+1))%mod;
	}
	
	hSuf[s.size()-1] = s[s.size()-1]-minimoCaracter+1;
	
	for(int i = s.size()-2; i>=0; i--){
		hSuf[i] = ( (hSuf[i+1]*B )% mod+(s[i]-minimoCaracter+1))%mod;
	}
}

int main(){
	pot[0]= 1;
	for(int i = 1; i<MAX;i++){
		pot[i] = (pot[i-1]*B)%mod;
	}
	cin >> s;
	precalc(s);
	return 0;
}
