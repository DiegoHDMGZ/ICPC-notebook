#include <bits/stdc++.h>
#define FOR(x) for(int i = 0; i < x ; i++)
#define MAX 100000
using namespace std;

typedef long long ll;
vector<ll> b(MAX); //b[i] es la longitud del borde más grande en la subcadena s[0...i]

void prefixFunction(string s){ //O(n)
	ll n = s.size();
	
	for(int i = 1; i < n; i++){
		ll bordeAnterior = b[i-1];
		while(bordeAnterior >0 && s[i] != s[bordeAnterior]/*el que le sigue al borde*/ ){
			bordeAnterior = b[bordeAnterior-1];
		}
		
		if(s[i] == s[bordeAnterior]){
			bordeAnterior++;
		}
		
		b[i] = bordeAnterior;
	}
}

vector<ll> ans; //posiciones de la cadena en donde empieza una ocurrencia de un patrón

void kmp(string p, string t){ //O(n+m)
//determina si el patron p aparece en t y enumera las posiciones de ocurrencia
	ll n = p.size();
	ll m = t.size();
	
	string s = p + "#" + t;
	
	for(int i = 1; i < s.size();i++){
		ll j = b[i-1];
		while(j > 0 && s[i] != s[j]){
			j = b[j-1];
		}
		
		if(s[i] == s[j]){
			j++;
		}
		
		b[i] = j;
		
		if(b[i] == n){
			ans.push_back(i- (n+1)-n+1);
		}
	}	
}
