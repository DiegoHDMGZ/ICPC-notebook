#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define ll long long
using namespace std;

const ll N = 1e5; //Suma de caracteres de todas las palabras + 1 : cantidad maxima de nodos
const ll ALPH = 26; // tamaño del alfabeto
ll trie[N][ALPH];
bool term[N]; //indica si es nodo terminal
ll nodos = 1;

void init(ll cnt){
	nodos = 1;
	for(int i = 0; i < cnt; i++){
		term[i] = 0;
		for(int j = 0; j < ALPH; j++){
			trie[i][j] = 0;
		}
	}
}

void addWord(string &s){
	int sz = s.size();
	int u = 0;
	
	for(int i = 0; i < sz; i++){
		int c = s[i] - 'a';
		if(trie[u][c] == 0){
			trie[u][c] = nodos++;
		}
		u = trie[u][c];
	}
	term[u] = true;
}

bool isPrefix(string &s){
	int sz = s.size();
	int u = 0;
	
	for(int i = 0; i < sz; i++){
		int c = s[i] - 'a';
		if(trie[u][c] == 0){
			return false;
		}
		u = trie[u][c];
	}
	return true;
}

bool isWord(string &s){
	int sz = s.size();
	int u = 0;
	
	for(int i = 0; i < sz; i++){
		int c = s[i] - 'a';
		if(trie[u][c] == 0){
			return false;
		}
		u = trie[u][c];
	}
	return term[u]; //validar si es nodo terminal
}
