#include <bits/stdc++.h>
#define debug(x) cout << #x << " = " << x << endl
#define ll long long
#define FOR(i,n) for(int i = 0; i < n; i++)
#define pb push_back
#define ms(a,valor,n) FOR(i,n) a[i] = valor;
#define snum(x) scanf("%lld",&x)
#define MAX (int)(1e6)

using namespace std;

const ll loga = (int)log2(MAX)+1;

struct sparseTable{
	ll st[MAX][loga];
	
	ll f(ll a, ll b){
		return a + b;
	}

	ll f(ll a){
		return a;
	}
	
	void precalculo(ll A[], ll n){ // O(n log n)
		for(int i = 0; i < n; i++){
			st[i][0] = f(A[i]);
		}
		
		for(int j = 1; (1 << j) <= n; j++){
			for(int i = 0; i + (1 << j) <= n; i++){
				st[i][j] = f(st[i][j-1], st[i + (1 << (j-1))][j-1]);
			}
		}
	}
	
	ll query(ll L, ll R){ //O(logn)
		ll ans = 0;
		ll T = R-L+1;
		ll lg = 31 - (__builtin_clz(T)); //te dice cuántos ceros hay a la izquierda
		
		for(int j = lg; j >= 0; j--){
			if((1 << j) <= T){
				ans = f(ans, st[L][j]);
				L += (1 << j);
				T -= (1 << j);
			}
		}
		
		return ans;
	}
	
	ll queryFast(ll L, ll R){ //O(1)
		//solo casos especiales que cuando se sobrelapen rangos, salga la misma respuesta ( como en min, max, gcd)
		ll T = R - L + 1;
		ll lg = 31 - (__builtin_clz(T));
		return f(st[L][lg], st[R- (1 << lg) + 1][lg]);
	}
	
}st;
