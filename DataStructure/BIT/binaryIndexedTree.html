#include <bits/stdc++.h>
#define MAX 100000

typedef long long ll;
// Binary Indexed Tree
// tree[i] = f[i - 2^r + 1] + ... + f[i]
//ultimo bit encendido = x & -x;

using namespace std;

struct FenwickTree{
	ll tree[MAX+1];
	
	FenwickTree(){
		for(ll i = 0; i<= MAX; i++){
			tree[i] = 0;
		}
	}
	
	void build(ll a[], ll n){ //O(nlogn)
		for(ll i = 0; i < n; i++){
			update(i+1, a[i]);
		}
	}
	
	ll query(ll i){ //O(log n)
		ll sum = 0;
		while(i > 0){
			sum += tree[i];
			i -= (i & -i);
		}
		return sum;
	}
	
	void update(ll i , ll delta){ //O(log MAX)
		while(i <= MAX){
			tree[i] += delta;
			i += (i & -i);
		}
	}		
} ft;
