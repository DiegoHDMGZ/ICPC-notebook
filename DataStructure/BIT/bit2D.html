#include <bits/stdc++.h>
#define MAXX 1000
#define MAXY 1000
#define ll long long
using namespace std;

struct FenwickTree2D{
	ll tree[MAXX+1][MAXY+1];
	
	FenwickTree2D(){ //O(MAXX * MAXY)
		for(int i = 0; i<= MAXX; i++){
			for(int j = 0; j <= MAXY; j++){
				tree[i][j] = 0;	
			}
		}
	}
	
	void build(){ //O(MAXX * MAXY)
		for(int i = 0; i<= MAXX; i++){
			for(int j = 0; j <= MAXY; j++){
				tree[i][j] = 0;	
			}
		}
	}
	
	ll query(int i, int j){ // O(log i * log j)
		ll sum = 0;
		while(i > 0){
			int _j = j;
			while(_j > 0){
				sum += tree[i][_j];
				_j -= (_j & -_j);
			}
			i -= (i & -i);
		}
		return sum;
	}
	
	ll query(ll x1, ll y1,ll x2, ll y2){
		return query(x2,y2)-query(x2,y1-1)-query(x1-1,y2)+query(x1-1,y1-1);
	}
	
	void update(int i, int j, int delta){ //O(log MAXX * log MAXY)
		while(i <= MAXX){
			int _j = j;
			while(_j <= MAXY){
				tree[i][_j] += delta;
				_j += (_j & -_j);
			}
			i += (i & -i);
		}
	}		
}ft;
